/**
  * Return all User, Activity Overtime date & type, with a PENDING status By an Invoice Manager ID
  * @param invoiceManagerId Id of the Invoice Manager
  * @param d Date begin limit
  * @return List of EFX_Tock__c related to InvoiceManagerId
  */
public List<EFX_Tock__c> getOvertimeApplicantsByInvoiceManager(Id invoiceManagerId, Datetime d) {

    return [
        SELECT BusinessActivity__r.Invoice_Manager__c, CreatedBy.Name, CreatedDate, Activity_Full_Name__c, Overtime_type__c, Overtime_Start_hour__c, Overtime_End_hour__c
        FROM EFX_Tock__c
        WHERE Overtime_type__c != NULL
        AND BusinessActivity__r.Invoice_Manager__c = : invoiceManagerId
        AND Status__c =  :PENDING_OVERTIME_STATUS
        AND RecordTypeId = :overtimeRecordTypeId
        AND CreatedDate >= :d
        ORDER BY CreatedDate
    ];

}
